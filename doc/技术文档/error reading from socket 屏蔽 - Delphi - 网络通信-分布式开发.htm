procedure TfmLogin.ApplicationEvents1Exception(Sender: TObject;
  E: Exception);
begin
  if E.ClassType.ClassName = 'ESocketConnectionError' then
  showmessage(E.ClassType.ClassName)
  else
  raise E;
end;
